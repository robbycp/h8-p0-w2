    ELSE IF year module to 400 not equal to 0
        THEN DISPLAY "Tahun Kabisat"
    ELSE
        THEN DISPLAY "Bukan Tahun Kabisat"
ELSE

3. Laundry Day
STORE total-pakaian to 20
STORE pakaian to 0

WHILE pakaian <= total-pakaian
    ADD pakaian by 1
    IF "pakaian" equals to "total pakaian"
        START the washing machine
    ELSE 
        DO NOT START the washing machine
